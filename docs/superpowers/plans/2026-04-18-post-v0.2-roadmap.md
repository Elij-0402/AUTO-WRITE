```ts
// src/lib/diff/text-diff.ts
import { diffChars } from 'diff'

export interface DiffSegment {
  type: 'equal' | 'insert' | 'delete'
  text: string
}

/**
 * Character-level diff between two plain-text strings.
 * Uses jsdiff's `diffChars` — works for Chinese since CJK has no word boundaries
 * that `diffWords*` can leverage; word-level diff degrades to line-level on
 * unsegmented CJK text.
 */
export function computeDiff(before: string, after: string): DiffSegment[] {
  const changes = diffChars(before, after)
  return changes.map(c => ({
    type: c.added ? 'insert' : c.removed ? 'delete' : 'equal',
    text: c.value,
  }))
}
```

```ts
// src/lib/diff/apply-ai-draft.ts
import type { InkForgeProjectDB } from '@/lib/db/project-db'
import type { EditorHandle } from '@/components/editor/editor-types'
import { createRevision } from '@/lib/db/revisions'

export function splitIntoParagraphs(text: string): string[] {
  return text
    .split(/\n\s*\n/)
    .map(p => p.replace(/^\s+|\s+$/g, ''))
    .filter(p => p.length > 0)
}

export function textToTiptapDoc(text: string): object {
  const paragraphs = splitIntoParagraphs(text)
  if (paragraphs.length === 0) {
    return { type: 'doc', content: [{ type: 'paragraph' }] }
  }
  return {
    type: 'doc',
    content: paragraphs.map(p => ({
      type: 'paragraph',
      content: [{ type: 'text', text: p }],
    })),
  }
}

export interface ApplyAIDraftParams {
  editorHandle: EditorHandle
  projectId: string
  chapterId: string
  /** Full replacement Tiptap document. */
  newDoc: object
  /** Optional label for the revision (e.g., "AI 草稿 - 第 3 章生成"). */
  revisionLabel?: string
}

/**
 * Unified acceptance path for AI-produced chapter content.
 * 1. Creates an `ai-draft` revision snapshot BEFORE mutation so the user can
 *    roll back via the history drawer.
 * 2. Calls editor ref's setContent to replace the live document.
 *
 * NOTE: persistence of the new content to the chapter row happens through
 * the editor's autosave loop — setContent fires onChange, use-chapter-editor
 * debounces and writes to Dexie.
 */
export async function applyAIDraft(
  db: InkForgeProjectDB,
  params: ApplyAIDraftParams
): Promise<void> {
  await createRevision(db, {
    projectId: params.projectId,
    chapterId: params.chapterId,
    snapshot: params.newDoc,
    source: 'ai-draft',
    label: params.revisionLabel,
  })
  params.editorHandle.setContent(params.newDoc)
}
```

```tsx
  onReject: () => void

  disabled?: boolean

  content,
  onAccept,
  onReject,
  onRegenerate,
  disabled = false,

    <div className="rounded-[var(--radius-card)] border border-[hsl(var(--accent-amber))]/40 bg-[hsl(var(--surface-1))] overflow-hidden">
```

```tsx
'use client'

import { useMemo } from 'react'
import { Check, X, RefreshCw } from 'lucide-react'
import { computeDiff, type DiffSegment } from '@/lib/diff/text-diff'
import { Button } from '@/components/ui/button'
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from '@/components/ui/dialog'

interface DiffViewProps {
  before: string
  after: string
  open: boolean
  onAccept: () => void
  onReject: () => void
  onRegenerate?: () => void
  disabled?: boolean
  title?: string
}

export function DiffView({
  before,
  after,
  open,
  onAccept,
  onReject,
  onRegenerate,
  disabled,
  title = 'AI 草稿预览',
}: DiffViewProps) {
  const segments = useMemo(() => computeDiff(before, after), [before, after])
  const stats = useMemo(() => summarize(segments), [segments])

  return (
    <Dialog open={open} onOpenChange={o => { if (!o) onReject() }}>
      <DialogContent className="max-w-3xl max-h-[80vh] flex flex-col">
        <DialogHeader>
          <DialogTitle className="text-lg">{title}</DialogTitle>
          <p className="text-[12px] text-muted-foreground">
            +{stats.insertions} 字 · -{stats.deletions} 字
          </p>
        </DialogHeader>

        <div className="flex-1 overflow-y-auto border rounded-[var(--radius-control)] p-4 bg-[hsl(var(--surface-1))]">
          <p className="whitespace-pre-wrap text-[14px] leading-[1.9] font-serif">
            {segments.map((seg, i) => (
              <SegmentSpan key={i} segment={seg} />
            ))}
          </p>
        </div>

        <DialogFooter className="gap-2">
          {onRegenerate && (
            <Button variant="ghost" size="sm" onClick={onRegenerate} disabled={disabled}>
              <RefreshCw className="w-3.5 h-3.5 mr-1" />
              重新生成
            </Button>
          )}
          <Button variant="ghost" size="sm" onClick={onReject} disabled={disabled}>
            <X className="w-3.5 h-3.5 mr-1" />
            拒绝
          </Button>
          <Button size="sm" onClick={onAccept} disabled={disabled}>
            <Check className="w-3.5 h-3.5 mr-1" />
            接受
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}

function SegmentSpan({ segment }: { segment: DiffSegment }) {
  if (segment.type === 'insert') {
    return (
      <span className="bg-[hsl(var(--accent-jade))]/20 text-[hsl(var(--accent-jade))] rounded-sm px-0.5 diff-segment-insert">
        {segment.text}
      </span>
    )
  }
  if (segment.type === 'delete') {
    return (
      <span className="bg-[hsl(var(--accent-coral))]/15 text-[hsl(var(--accent-coral))] line-through rounded-sm px-0.5 diff-segment-delete">
        {segment.text}
      </span>
    )
  }
  return <span className="text-foreground/90">{segment.text}</span>
}

function summarize(segments: DiffSegment[]) {
  let insertions = 0
  let deletions = 0
  for (const s of segments) {
    if (s.type === 'insert') insertions += s.text.length
    else if (s.type === 'delete') deletions += s.text.length
  }
  return { insertions, deletions }
}
```

```ts
// src/lib/analysis/arc-map.ts
import type { InkForgeProjectDB } from '@/lib/db/project-db'
import type { AIClientConfig } from '@/lib/ai/client'
import { streamChat } from '@/lib/ai/client'
import { EMIT_ARC_MAP_SCHEMA, type EmitArcMapInput } from '@/lib/ai/tools/schemas'
import { extractPlainText, contentHash } from './content-hash'

export interface GenerateArcMapParams {
  config: AIClientConfig
  signal?: AbortSignal
}

const ARC_SYSTEM_PROMPT = `你是中文网文剧情结构分析师。基于用户提供的章节摘要序列,分析每章的剧情节拍、张力、情绪,输出结构化 arc map。

【规则】
- 必须为每个给定 chapterId 输出一条分析。
- tension 评分 0-10,按整体故事均值 5 为基准。
- emotion 评分 -1 到 1,反映该章总体情绪色调。
- keyEvents 每条不超过 15 字,只提取推动剧情的核心事件,不叙述细节。
- summary 30 字以内,概括章节核心。
- overallArc 120 字以内,概括整体弧线走向。

【输出】
必须通过 emit_arc_map 工具一次性返回全部结果。不要输出自由文本。`

export async function generateArcMap(
  db: InkForgeProjectDB,
  projectId: string,
  params: GenerateArcMapParams
): Promise<EmitArcMapInput> {
  if (params.config.provider !== 'anthropic') {
    throw new Error('剧情弧线分析需要 Anthropic provider(其他 provider 不支持结构化工具调用)')
  }

  const chapters = await db.chapters
    .where({ projectId })
    .filter(c => !c.deletedAt)
    .sortBy('order')

  if (chapters.length === 0) {
    return { chapters: [], overallArc: '暂无章节内容,无法分析。' }
  }

  const texts = chapters.map(c => extractPlainText(c.content)).filter(Boolean)
  const currentHash = contentHash(texts)

  const cached = await db.analyses.where({ kind: 'arc-map' }).first()
  if (cached && cached.invalidationKey === currentHash) {
    return cached.result as EmitArcMapInput
  }

  // Build the user message with numbered chapter summaries
  const chapterBlock = chapters
    .map((c, idx) => {
      const text = extractPlainText(c.content).slice(0, 800)
      return `### 章节 ${idx + 1} (chapterId: ${c.id})
标题:${c.title}
正文节选:
${text || '(空章节)'}
`
    })
    .join('\n')

  const userMessage = `请分析以下 ${chapters.length} 个章节的剧情弧线:\n\n${chapterBlock}`

  const events = streamChat(params.config, {
    segmentedSystem: {
      baseInstruction: ARC_SYSTEM_PROMPT,
      worldBibleContext: '',
      runtimeContext: '',
    },
    messages: [{ role: 'user', content: userMessage }],
    signal: params.signal,
  })

  let result: EmitArcMapInput | null = null
  for await (const event of events) {
    if (event.type === 'tool_call' && event.name === EMIT_ARC_MAP_SCHEMA.name) {
      result = event.input as EmitArcMapInput
    } else if (event.type === 'error') {
      throw new Error(event.message)
    }
  }

  if (!result) {
    throw new Error('AI 未通过工具调用返回结果,请重试(建议切换至支持 tool use 的 Claude 模型)')
  }

  await db.analyses.put({
    id: 'arc-map',
    projectId,
    kind: 'arc-map',
    invalidationKey: currentHash,
    result,
    createdAt: new Date(),
  })

  return result
}
```

```ts
'use client'

import { useState, useCallback, useRef, useEffect } from 'react'
import { useAIConfig } from './use-ai-config'
import { useWorldEntries } from './use-world-entries'
import { useChapters } from './use-chapters'
import { streamChat } from '@/lib/ai/client'
import { searchRelevantEntries } from '@/lib/rag/search'
import { getDefaultEmbedder } from '@/lib/rag/default-embedder'
import { createProjectDB } from '@/lib/db/project-db'
import { trimToTokenBudget } from './use-context-injection'
import { recordUsage } from '@/lib/db/ai-usage-queries'
import type { AIUsageEvent } from '@/lib/db/project-db'

export interface GenerationState {
  status: 'idle' | 'generating' | 'complete' | 'error'
  streamingContent: string
  error: string | null
}

const GENERATION_BASE_INSTRUCTION = `你是 InkForge 的中文网文章节创作助手。根据作者提供的章节大纲和世界观百科,创作完整章节。

【规则】
- 内容丰富,情节完整,对白与动作描写均衡。
- 风格遵循中文网文节奏:节拍清晰、悬念感强、语言流畅。
- 严格依据【世界观百科】中已登记的设定,不要虚构未提及的关键设定。
- 章节结尾留有悬念或自然过渡到下一章。
- 用\\n\\n分段。每段自然成章,不要在一段里塞过多内容。
- 输出纯正文,不要加 markdown 标题或章节编号。`

export function useChapterGeneration(projectId: string, chapterId: string) {
  const { config } = useAIConfig(projectId)
  const { entries, entriesByType } = useWorldEntries(projectId)
  const { chapters } = useChapters(projectId)

  const [state, setState] = useState<GenerationState>({
    status: 'idle',
    streamingContent: '',
    error: null,
  })
  const abortRef = useRef<AbortController | null>(null)
  const entriesRef = useRef(entries)
  const entriesByTypeRef = useRef(entriesByType)
  useEffect(() => { entriesRef.current = entries }, [entries])
  useEffect(() => { entriesByTypeRef.current = entriesByType }, [entriesByType])

  const currentChapter = chapters.find(c => c.id === chapterId)

  const startGeneration = useCallback(async () => {
    if (!config.apiKey) {
      setState({ status: 'error', streamingContent: '', error: '请先配置 AI 设置:API Key 必填' })
      return
    }
    if (config.provider === 'openai-compatible' && !config.baseUrl) {
      setState({ status: 'error', streamingContent: '', error: '请先配置 AI 设置:OpenAI 兼容模式需要 Base URL' })
      return
    }
    if (!currentChapter) {
      setState({ status: 'error', streamingContent: '', error: '章节未找到' })
      return
    }
    const { title, outlineSummary, outlineTargetWordCount } = currentChapter
    if (!outlineSummary) {
      setState({ status: 'error', streamingContent: '', error: '请先填写章节大纲摘要' })
      return
    }

    setState({ status: 'generating', streamingContent: '', error: null })

    const db = createProjectDB(projectId)
    const embedder = getDefaultEmbedder()
    const relevant = entriesRef.current
      ? await searchRelevantEntries({
          db, projectId, embedder, query: outlineSummary,
          entries: entriesRef.current, entriesByType: entriesByTypeRef.current, topK: 20,
        })
      : []
    const trimmed = trimToTokenBudget(relevant, 4000)

    const worldBibleBlock = trimmed.length === 0
      ? '【世界观百科】\n(暂无相关条目)'
      : `【世界观百科】\n${trimmed.map(e => `- ${e.type} · ${e.name}: ${('description' in e && e.description) || ''}`).join('\n')}`

    const runtimeBlock = `【本章任务】
标题:${title}
大纲:${outlineSummary}
目标字数:${outlineTargetWordCount ? `${outlineTargetWordCount} 字` : '未设定'}`

    abortRef.current = new AbortController()
    const startedAt = Date.now()
    let fullText = ''
    let inputTokens = 0, outputTokens = 0, cacheReadTokens = 0, cacheWriteTokens = 0

    try {
      const events = streamChat(
        { provider: config.provider, apiKey: config.apiKey, baseUrl: config.baseUrl, model: config.model },
        {
          segmentedSystem: {
            baseInstruction: GENERATION_BASE_INSTRUCTION,
            worldBibleContext: worldBibleBlock,
            runtimeContext: runtimeBlock,
          },
          messages: [{ role: 'user', content: '请基于以上大纲和世界观背景,创作完整章节。' }],
          signal: abortRef.current.signal,
        }
      )

      for await (const event of events) {
        if (event.type === 'text_delta') {
          fullText += event.delta
          setState(s => ({ ...s, streamingContent: fullText }))
        } else if (event.type === 'usage') {
          if (event.inputTokens !== undefined) inputTokens = event.inputTokens
          if (event.outputTokens !== undefined) outputTokens = event.outputTokens
          if (event.cacheReadTokens !== undefined) cacheReadTokens = event.cacheReadTokens
          if (event.cacheWriteTokens !== undefined) cacheWriteTokens = event.cacheWriteTokens
        } else if (event.type === 'error') {
          throw new Error(event.message)
        }
      }

      setState({ status: 'complete', streamingContent: fullText, error: null })
    } catch (err) {
      if (err instanceof Error && err.name === 'AbortError') {
        setState({ status: 'idle', streamingContent: '', error: null })
      } else {
        setState(s => ({ ...s, status: 'error', error: err instanceof Error ? err.message : '生成失败' }))
      }
    } finally {
      abortRef.current = null
      if (inputTokens > 0 || outputTokens > 0) {
        const usage: AIUsageEvent = {
          id: crypto.randomUUID(),
          projectId,
          conversationId: `generation:${chapterId}`,
          kind: 'generation',
          provider: config.provider,
          model: config.model ?? '',
          inputTokens, outputTokens, cacheReadTokens, cacheWriteTokens,
          latencyMs: Date.now() - startedAt,
          createdAt: Date.now(),
        }
        await recordUsage(db, usage).catch(e => console.warn('[useChapterGeneration] recordUsage failed:', e))
      }
    }
  }, [config, currentChapter, projectId, chapterId])

  const cancelGeneration = useCallback(() => {
    abortRef.current?.abort()
  }, [])

  const resetGeneration = useCallback(() => {
    setState({ status: 'idle', streamingContent: '', error: null })
  }, [])

  return {
    ...state,
    startGeneration,
    cancelGeneration,
    resetGeneration,
  }
}
```

```ts
// src/lib/rag/bge-embedder.ts
import type { Embedder } from './embedder'

export class BGEEmbedder implements Embedder {
  readonly dim = 512
  readonly id = 'bge-small-zh-v1.5'
  private pipelinePromise: Promise<(texts: string[]) => Promise<unknown>> | null = null

  private async getPipeline() {
    if (!this.pipelinePromise) {
      this.pipelinePromise = (async () => {
        const { pipeline } = await import('@xenova/transformers')
        const p = await pipeline('feature-extraction', 'Xenova/bge-small-zh-v1.5', {
          quantized: true,
        })
        return async (texts: string[]) => p(texts, { pooling: 'mean', normalize: true })
      })()
    }
    return this.pipelinePromise
  }

  async embed(texts: string[]): Promise<Float32Array[]> {
    if (texts.length === 0) return []
    const run = await this.getPipeline()
    const output = (await run(texts)) as { data: Float32Array; dims: [number, number] }
    const [count, dim] = output.dims
    const result: Float32Array[] = []
    for (let i = 0; i < count; i++) {
      const slice = output.data.slice(i * dim, (i + 1) * dim)
      // Model already normalizes when normalize: true, but we re-normalize to
      // be defensive — tests depend on L2 ≈ 1.
      result.push(l2Normalize(new Float32Array(slice)))
    }
    return result
  }
}

function l2Normalize(v: Float32Array): Float32Array {
  let sumSq = 0
  for (let i = 0; i < v.length; i++) sumSq += v[i] * v[i]
  if (sumSq === 0) return v
  const inv = 1 / Math.sqrt(sumSq)
  for (let i = 0; i < v.length; i++) v[i] *= inv
  return v
}
```

```ts
export type EmbedderKind = 'hashgram' | 'bge'

const STORAGE_KEY = 'inkforge-embedder'

export function getEmbedderPreference(): EmbedderKind {
  if (typeof window === 'undefined' || typeof localStorage === 'undefined') {
    return 'hashgram'
  }
  const raw = localStorage.getItem(STORAGE_KEY)
  if (raw === 'bge' || raw === 'hashgram') return raw
  return 'hashgram'
}

export function setEmbedderPreference(kind: EmbedderKind): void {
  if (typeof window === 'undefined' || typeof localStorage === 'undefined') return
  localStorage.setItem(STORAGE_KEY, kind)
```

```ts
/**
 * Default embedder selection. Reads user preference from localStorage and
 * instantiates either the always-available HashGramEmbedder or the BGE
 * neural embedder (lazy-loaded when the user has opted in).
 *
 * BGE instance creation is synchronous (the pipeline itself loads lazily on
 * first .embed() call), so the selector stays zero-cost for callers.
 */

import { HashGramEmbedder, type Embedder } from './embedder'
import { BGEEmbedder } from './bge-embedder'
import { getEmbedderPreference } from './embedder-preference'

let cached: Embedder | null = null
let cachedKind: 'hashgram' | 'bge' | null = null

export function getDefaultEmbedder(): Embedder {
  const pref = getEmbedderPreference()
  if (cached && cachedKind === pref) return cached

  cached = pref === 'bge' ? new BGEEmbedder() : new HashGramEmbedder()
  cachedKind = pref
  return cached
}

/** For tests only — resets the cached instance. */
export function __resetEmbedderForTests(): void {
  cached = null
  cachedKind = null
}
```

```ts
// scripts/benchmark-rag.ts
/**
 * RAG recall benchmark. Runs 20 Chinese referential queries against a fixture
 * world-bible project and reports Top-5 recall for hash-gram vs BGE.
 *
 * Run: `pnpm tsx scripts/benchmark-rag.ts`
 */

import 'fake-indexeddb/auto'
import { createProjectDB } from '../src/lib/db/project-db'
import { indexWorldEntries } from '../src/lib/rag/indexer'
import { searchRelevantEntries } from '../src/lib/rag/search'
import { HashGramEmbedder } from '../src/lib/rag/embedder'
import { BGEEmbedder } from '../src/lib/rag/bge-embedder'

interface TestCase {
  query: string
  expectedEntryIds: string[]  // Top-5 should include at least one of these
}

const FIXTURE_ENTRIES = [
  { id: 'e1', type: 'character', name: '林清霜', description: '女主,仙宗剑修,师从玄一道人' },
  { id: 'e2', type: 'character', name: '玄一道人', description: '男主的师父,掌门' },
  { id: 'e3', type: 'character', name: '沈墨舟', description: '男主,魔道少主' },
  // ... 加到 30 条以上,每类型约 8 条
] as const

const TEST_CASES: TestCase[] = [
  { query: '她的师父是谁', expectedEntryIds: ['e2'] },
  { query: '女主来自什么门派', expectedEntryIds: ['e1'] },
  { query: '男主有什么背景', expectedEntryIds: ['e3'] },
  // ... 20 条
]

async function runBench(kind: 'hashgram' | 'bge') {
  const db = createProjectDB('bench')
  await db.worldEntries.bulkAdd(FIXTURE_ENTRIES.map(e => ({
    ...e, projectId: 'bench', createdAt: new Date(), updatedAt: new Date(), deletedAt: null,
  })) as any)
  const entries = await db.worldEntries.where({ projectId: 'bench' }).toArray() as any[]
  const entriesByType = entries.reduce((acc: any, e) => {
    (acc[e.type] ??= []).push(e)
    return acc
  }, {})

  const embedder = kind === 'bge' ? new BGEEmbedder() : new HashGramEmbedder()
  await indexWorldEntries(db, { projectId: 'bench', entries, embedder })

  let hits = 0
  for (const tc of TEST_CASES) {
    const top = await searchRelevantEntries({
      db, projectId: 'bench', embedder,
      query: tc.query, entries, entriesByType, topK: 5,
    })
    if (top.some(e => tc.expectedEntryIds.includes(e.id))) hits++
  }
  return { kind, recall: hits / TEST_CASES.length, hits, total: TEST_CASES.length }
}

async function main() {
  const hash = await runBench('hashgram')
  const bge = await runBench('bge')
  console.log(JSON.stringify({ hashgram: hash, bge }, null, 2))
}

main().catch(e => { console.error(e); process.exit(1) })
```
